import { resource } from '@angular/core';

jokes = resource<Joke[], {}>({
  params: () => ({}),
  loader: async ({ abortSignal }) => {
    const res = await fetch('https://api.example.com/jokes',
      { signal: abortSignal });
    return res.json() as Promise<Joke[]>;
  },
});
