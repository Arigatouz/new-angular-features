import { resource, signal } from '@angular/core';

// stream: synchronous → resource resolves immediately with [].
// EventSource fires onmessage → signal updates → resource.value() updates.
wikiEdits = resource({
  stream: ({ abortSignal }) => {
    const s = signal({ value: [] });
    const src = new EventSource(WIKI_STREAM_URL);
    abortSignal.addEventListener('abort', () => src.close(), { once: true });
    src.onmessage = (e) =>
      s.update(prev => ({
        value: [JSON.parse(e.data), ...prev.value].slice(0, 20)
      }));
    return s;  // Angular watches this Signal forever
  },
});
