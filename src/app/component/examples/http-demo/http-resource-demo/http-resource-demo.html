import { httpResource } from '@angular/common/http';

jokes = httpResource<Joke[]>(
  () => 'https://api.example.com/jokes',
  { defaultValue: [] },
);

// Reactive state signals
jokes.isLoading()  // boolean
jokes.error()      // HttpErrorResponse | undefined
jokes.value()      // Joke[] (defaultValue keeps it defined)
jokes.reload()     // manually refetch
