  params: () => (count() !== undefined ? { count: count()! } : undefined),

    return timer(2000).pipe(
      concatMap(() => this.#http.get<Joke[]>(`/jokes/random/${params.count}`)),
      takeUntil(this.#manualCancel$), // "Cancel" button
    );
  },
});
